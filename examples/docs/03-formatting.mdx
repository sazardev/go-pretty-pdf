## Code Blocks

```go
package main

import (
    "context"
    "fmt"
    "github.com/sazardev/go-pretty-pdf"
)

func main() {
    pdf, err := prettypdf.New(
        prettypdf.WithSourceDir("./docs"),
        prettypdf.WithOutputFile("output.pdf"),
        prettypdf.WithTitle("My Document"),
    )
    if err != nil {
        panic(err)
    }
    err = pdf.Build(context.Background())
    if err != nil {
        fmt.Printf("Error: %v\n", err)
    }
}
```
